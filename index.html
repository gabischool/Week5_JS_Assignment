Task 3: Top Scorers Leaderboard 🏆⚽

Task Description: Update and print a leaderboard of top scorers.

        
const topScorers = [
  { name: "Messi", score: 5 },
  { name: "Ronaldo", score: 3 },
  { name: "Neymar", score: 4 }
];

function updateScore(playerName, scoreToAdd) {
  for (let i = 0; i < topScorers.length; i++) {
    if (topScorers[i].name === playerName) {
      topScorers[i].score += scoreToAdd;
      return;
}
}

topScorers.push({ name: playerName, score: scoreToAdd });
}

function printLeaderboard() {
  topScorers.sort((a, b) => b.score - a.score);
  for (let i = 0; i < topScorers.length; i++) {
    console.log(i + 1.{topScorers[i].name}: ${topScorers[i].score});
  }
}

updateScore("Ronaldo", 2);
updateScore("Nuno", 6);
updateScore("Ronaldo", 2);
updateScore("Messi", 1);
printLeaderboard();

Expected Output in Console:

1. Nuno: 6
2. Messi: 6
3. Ronaldo: 5
4. Neymar: 4
